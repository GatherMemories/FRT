    private boolean hasFilesToDelete(FileNode node) {
        if (node == null) {
            return false;
        }
        if (node.isFile()) {
            return true;
        }
        for (FileNode child : node()) {
            if (hasFilesToDelete(child)) {
                return true;
            }
        }
        return false;
    }

    private void printDeleteFiles(FileNode node, OperationContext context, int depth) {

        String indent = "  ".repeat(depth);

        if (node.isFile()) {
            Path relativePath = context.getRelativePath(node.getPath());

        } else {
            System.out.println(indent + "ğŸ“ " + node.getName() + "/");

            }
        }
    }

    private int countFiles(FileNode node) {
        if (node == null) {
            return 0;
        }
        if (node.isFile()) {
            return 1;
        }
        int count = 0;
        for (FileNode child : node.getChildren()) {
            count += countFiles(child);
        }
        return count;
    }
